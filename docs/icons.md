# SVG Icons

## How to include an svg icon

```js
// some-element.js

const html = require('choo/html')
const icon = require('@justifay/icon-element')

function someElement () {
  return html`
    <div>
      ${icon('logo', { 'class':'icon icon--md fill-black'})}
    </div>
}

module.exports = someElement

```
